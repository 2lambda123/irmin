Module Irmin_unix.Irmin_fs.RW

module RW: Irmin.RW_MAKER

Read-write stores


include Irmin.RO
val update : t -> key -> value -> unit Lwt.t
Replace the contents of key by value if key is already defined and create it otherwise.
val remove : t -> key -> unit Lwt.t
Remove the given key.
val watch : t -> key -> value option Lwt_stream.t
Watch the stream of values associated to a given key. Return None if the value is removed.
