include Irmin.RW
type watch
The type of watch handlers.
val watch_key : t ->
       key ->
       ?init:value -> (value Irmin.diff -> unit Lwt.t) -> watch Lwt.t
watch_key t k ?init f adds f to the list of t's watch handlers for the key k and returns the watch handler to be used with Irmin.RRW.unwatch. init is the optional initial value of the key.
val watch : t ->
       ?init:(key * value) list ->
       (key -> value Irmin.diff -> unit Lwt.t) -> watch Lwt.t
watch t ?init f adds f to the list of t's watch handlers and returns the watch handler to be used with Irmin.RRW.unwatch. init is the optional initial values. It is more efficient to use Irmin.RRW.watch_key to watch only a single given key.
val unwatch : t -> watch -> unit Lwt.t
unwatch t w removes w from t's watch handlers.
