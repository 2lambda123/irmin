Module Irmin.S.Branch

module Branch: sig .. end
Manipulate branches.

Branch Store

Manipulate relations between branches and commits.

val mem : Irmin.S.repo -> Irmin.S.branch -> bool Lwt.t
mem r b is true iff b is present in r.
val find : Irmin.S.repo -> Irmin.S.branch -> Irmin.S.commit option Lwt.t
find r b is Some c iff c is bound to b in t. It is None if b is not present in t.
val get : Irmin.S.repo -> Irmin.S.branch -> Irmin.S.commit Lwt.t
get t b is similar to Irmin.S.Branch.find but raise Invalid_argument if b is not present in t.
val set : Irmin.S.repo -> Irmin.S.branch -> Irmin.S.commit -> unit Lwt.t
set t b c bounds c to b in t.
val remove : Irmin.S.repo -> Irmin.S.branch -> unit Lwt.t
remove t b removes b from t.
val list : Irmin.S.repo -> Irmin.S.branch list Lwt.t
list t is the list of branches present in t.
val watch : Irmin.S.repo ->
       Irmin.S.branch ->
       ?init:Irmin.S.commit ->
       (Irmin.S.commit Irmin.diff -> unit Lwt.t) -> Irmin.S.watch Lwt.t
watch t b f calls f on every change in b.
val watch_all : Irmin.S.repo ->
       ?init:(Irmin.S.branch * Irmin.S.commit) list ->
       (Irmin.S.branch -> Irmin.S.commit Irmin.diff -> unit Lwt.t) ->
       Irmin.S.watch Lwt.t
watch_all t f calls f on every branch-related change in t, including creation/deletion events.
