Module type Irmin.Backend.REMOTE

module type REMOTE = functor (S : Ir_bc.STORE) -> sig .. end
Parameters:
S : Ir_bc.STORE 
val fetch : Irmin.S.t -> ?depth:int -> string -> Irmin.S.head option Lwt.t
fetch t uri fetches the contents of the remote store located at uri into the local store t. Return None if the remote store is empty, otherwise, return the head of uri.
val push : Irmin.S.t -> ?depth:int -> string -> Irmin.S.head option Lwt.t
push t uri pushes the contents of the local store t into the remote store located at uri. Return None is the local store is empty, otherwise, return the head of t.
